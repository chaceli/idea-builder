### 2.5 场景视图（Scenarios）

#### 场景 1：用户发送消息

```
参与者: 用户 → 小程序 → API → AI服务 → 数据库

1. 用户在小程序输入框输入文字
2. 小程序调用 POST /api/chat/message
3. API 接收请求，验证 Token
4. 调用 AIService.processMessage()
5. AIService:
   a. 检索记忆构建上下文
   b. 构建 Prompt
   c. 调用 MiniMax API
   d. 处理响应
6. 保存消息到数据库
7. 更新用户画像
8. 返回响应给小程序
9. 小程序展示 AI 回复
10. 播放语音（可选）
```

#### 场景 2：AI 主动关心

```
参与者: 定时任务 → 调度服务 → 用户选择 → 推送

1. 定时任务每晚 8 点检查:
   - 用户设置中"主动关心"是否开启
   - 距离上次对话是否超过 24 小时
2. 对需要关心的用户:
   a. 检索用户画像和记忆
   b. 构建关心 Prompt
   c. 调用 AI 生成关心消息
   d. 通过小程序订阅消息推送
3. 用户收到推送通知
4. 用户点击进入对话
```

## 3. 核心模块设计

### 3.1 AI 对话模块

```typescript
// src/backend/src/services/AIService.ts

interface AIMessage {
  role: 'system' | 'user' | 'assistant';
  content: string;
}

interface ChatRequest {
  userId: string;
  content: string;
  type: 'text' | 'voice';
  voiceUrl?: string;
}

interface ChatResponse {
  message: {
    id: string;
    role: 'assistant';
    content: string;
    voiceUrl?: string;
    emotion: 'happy' | 'sad' | 'neutral' | 'excited' | 'sympathy';
  };
}

class AIService {
  async processMessage(req: ChatRequest): Promise<ChatResponse> {
    // 1. 检索记忆
    const memories = await this.memoryService.retrieve(req.userId, req.content);
    
    // 2. 构建 Prompt
    const prompt = this.buildPrompt(memories, req.content);
    
    // 3. 调用 AI
    const aiResponse = await this.callAI(prompt);
    
    // 4. 情感分析
    const emotion = await this.emotionService.analyze(aiResponse);
    
    // 5. 语音合成（可选）
    let voiceUrl: string | undefined;
    if (req.type === 'voice') {
      voiceUrl = await this.ttsService.synthesize(aiResponse);
    }
    
    // 6. 保存对话
    await this.saveMessage(req, aiResponse);
    
    // 7. 更新记忆
    await this.memoryService.update(req.userId, aiResponse);
    
    return {
      message: {
        id: generateId(),
        role: 'assistant',
        content: aiResponse,
        voiceUrl,
        emotion,
      },
    };
  }
}
```

### 3.2 记忆系统模块

```typescript
// src/backend/src/services/MemoryService.ts

interface Memory {
  id: string;
  userId: string;
  type: 'short-term' | 'long-term' | 'fact';
  content: string;
  importance: number; // 1-5
  embedding?: number[];
  createdAt: Date;
  updatedAt: Date;
}

class MemoryService {
  // 检索记忆
  async retrieve(userId: string, query: string): Promise<Memory[]> {
    // 1. 获取短期记忆（最近 20 轮）
    const shortTerm = await this.getShortTerm(userId);
    
    // 2. 获取长期记忆（语义检索）
    const longTerm = await this.getLongTerm(userId, query);
    
    // 3. 合并排序
    return this.mergeAndSort(shortTerm, longTerm);
  }
  
  // 更新记忆
  async update(userId: string, newContent: string): Promise<void> {
    // 1. 提取关键信息
    const facts = this.extractFacts(newContent);
    
    // 2. 评估重要性
    for (const fact of facts) {
      const importance = await this.assessImportance(fact);
      
      if (importance >= 4) {
        // 重要：存入长期记忆
        await this.saveLongTerm(userId, fact, importance);
      } else {
        // 不重要：存入短期记忆
        await this.saveShortTerm(userId, fact);
      }
    }
  }
}
```
